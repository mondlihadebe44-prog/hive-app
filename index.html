create table comments ( id uuid default gen_random_uuid() primary key, user_id uuid references profiles(id) on delete cascade, post_id uuid references posts(id) on delete cascade, body text not null, created_at timestamp default now() ); create policy "Anyone can read comments" on comments for select to anon, authenticated using (true); create policy "Users can insert comments" on comments for insert to authenticated with check (auth.uid() = user_id);
